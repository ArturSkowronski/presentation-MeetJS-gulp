Plusy

Kod ponad konfiguracją

var gulp = require('gulp')       
gulp.task('default',  function() {
    gulp.run('sass');
    gulp.src('_scss/ *.scss').pipe(gulp.dest('css/'));
    gulp.watch(['_scss/**/ *.scss'], function(event) {
        gulp.run('sass');
    });
});

Przyjemne w użyciu API

  • gulp.task(name, fn)
  • gulp.run(tasks...)
  • gulp.watch(glob, fn)
  • gulp.src(glob)
  • gulp.dest(folder)
